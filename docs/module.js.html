/**
 * Calcule l'âge d'une personne
 *
 * @param {object} p - Objet personne avec une propriété birth (Date)
 * @return {number} L'âge en années
 */
function calculateAge(p) {
    if (p === undefined || p === null) {
        throw new Error('Argument is required');
    }

    if (typeof p !== 'object') {
        throw new Error('Argument must be an object');
    }

    if (!p.hasOwnProperty('birth')) {
        throw new Error('Object must have a birth property');
    }

    if (!(p.birth instanceof Date)) {
        throw new Error('Birth must be a Date object');
    }

    if (isNaN(p.birth.getTime())) {
        throw new Error('Birth date is invalid');
    }

    let dateDiff = new Date(Date.now() - p.birth.getTime());
    let age = Math.abs(dateDiff.getUTCFullYear() - 1970);
    return age;
}

module.exports = { calculateAge };
